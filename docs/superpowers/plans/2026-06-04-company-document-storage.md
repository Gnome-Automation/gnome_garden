# Company Document Storage (AshStorage + Garage) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add file upload support to the Company Documents feature using AshStorage, backed by Garage (S3-compatible, self-hosted) in production and local disk in dev.

**Architecture:** Mirror the existing `GnomeGarden.Acquisition.Document` AshStorage pattern exactly. Add `CompanyDocumentBlob` and `CompanyDocumentAttachment` resources, add the `AshStorage` extension to `CompanyDocument`, wire config for dev/test/prod, add a New Document upload modal to `DocumentsLive`, and add a download controller that streams files from the storage service. Email attachments in `DocumentEmail` are updated to download bytes from storage instead of reading a static path.

**Tech Stack:** Elixir/Phoenix, Ash Framework 3.x, AshStorage (already in mix.exs), AshPostgres, LiveView `allow_upload`, Garage (S3-compatible) via `AshStorage.Service.S3`

**Key reference files:**
- `lib/garden/acquisition/document_blob.ex` — blob resource to copy
- `lib/garden/acquisition/document_attachment.ex` — attachment resource to copy
- `lib/garden/acquisition/document.ex` — storage do block pattern
- `lib/garden_web/controllers/pi_document_controller.ex` — download pattern (`service.download(key, context)`)
- `lib/garden_web/live/acquisition/finding_document_live/form.ex` — `allow_upload` + `consume_uploaded_entries` pattern
- `config/runtime.exs` lines 77-95 — Garage S3 config pattern
- `config/test.exs` lines 38-39 — test storage config pattern

        :byte_size,
        :checksum,

def handle_event("open_upload_modal", _params, socket) do
  {:noreply,
   socket
   |> assign(:upload_modal_open, true)
   |> assign(:upload_name, "")
   |> assign(:upload_category, "tax")
   |> assign(:upload_version, "")
   |> assign(:upload_description, "")
   |> assign(:upload_error, nil)}

def handle_event("close_upload_modal", _params, socket) do
  {:noreply, assign(socket, :upload_modal_open, false)}
end

def handle_event("validate_upload", _params, socket) do
  {:noreply, socket}
end

def handle_event("save_document", %{"doc" => params}, socket) do
  entries = socket.assigns.uploads.file.entries

  if Enum.empty?(entries) do
    {:noreply, assign(socket, :upload_error, "Please select a PDF file to upload")}
  else
    result =
      consume_uploaded_entries(socket, :file, fn %{path: path}, entry ->
        file = %Ash.Type.File.Implementation.CachedFile{
          path: path,
          filename: entry.client_name,
          content_type: entry.client_type
        }

        Documents.create_document(%{
          name: params["name"],
          category: String.to_existing_atom(params["category"]),
          version: params["version"],
          description: if(params["description"] != "", do: params["description"]),
          status: :active,
          file: file
        })
      end)

    case List.first(result) do
      {:ok, _doc} ->
        {:ok, docs} = Documents.list_active_documents()

        {:noreply,
         socket
         |> assign(:upload_modal_open, false)
         |> assign(:all_docs, docs)
         |> apply_filters()
         |> put_flash(:info, "Document uploaded successfully")}

      {:error, reason} ->
        {:noreply, assign(socket, :upload_error, "Upload failed: #{inspect(reason)}")}

      nil ->
        {:noreply, assign(socket, :upload_error, "No file was processed")}
    end
  end
end
```

# lib/garden_web/controllers/company_document_download_controller.ex
defmodule GnomeGardenWeb.CompanyDocumentDownloadController do
  use GnomeGardenWeb, :controller

  alias GnomeGarden.Documents

  def show(conn, %{"id" => id}) do
    doc =
      Documents.get_document!(id,
        load: [file: [blob: []]],
        actor: conn.assigns.current_user
      )

    case doc do
      %{file: %{blob: blob}} when is_map(blob) ->
        stream_from_storage(conn, doc, blob)

      _ when is_binary(doc.file_path) ->
        # Legacy: static file in priv/static
        static_path = Path.join([:code.priv_dir(:gnome_garden), "static", doc.file_path])
        send_download(conn, {:file, static_path}, filename: Path.basename(doc.file_path))

      _ ->
        conn
        |> put_status(:not_found)
        |> text("File not found")
    end
  end

  defp stream_from_storage(conn, doc, blob) do
    service = blob.service_name

    configured_service =
      :gnome_garden
      |> Application.get_env(GnomeGarden.Documents.CompanyDocument, [])
      |> Keyword.get(:storage, [])
      |> Keyword.get(:service)

    service_opts =
      case configured_service do
        {^service, runtime_opts} ->
          persisted = blob.service_opts |> Enum.map(fn {k, v} -> {String.to_atom(k), v} end)
          Keyword.merge(persisted, runtime_opts)

        _ ->
          blob.service_opts |> Enum.map(fn {k, v} -> {String.to_atom(k), v} end)
      end

    context = AshStorage.Service.Context.new(service_opts)

    case service.download(blob.key, context) do
      {:ok, binary} ->
        content_type = blob.content_type || "application/octet-stream"
        filename = blob.filename || "#{doc.name}-#{doc.version}.pdf"

        conn
        |> put_resp_content_type(content_type)
        |> put_resp_header("content-disposition", ~s(attachment; filename="#{filename}"))
        |> send_resp(200, binary)

      {:error, _reason} ->
        conn
        |> put_status(:internal_server_error)
        |> text("Download failed")
    end
  end
end
```

  cond do
    # AshStorage-managed file
    match?(%{file: %{blob: blob}} when is_map(blob), document) ->
      blob = document.file.blob
      service = blob.service_name

      configured_service =
        :gnome_garden
        |> Application.get_env(GnomeGarden.Documents.CompanyDocument, [])
        |> Keyword.get(:storage, [])

      case service.download(blob.key, AshStorage.Service.Context.new(service_opts)) do
